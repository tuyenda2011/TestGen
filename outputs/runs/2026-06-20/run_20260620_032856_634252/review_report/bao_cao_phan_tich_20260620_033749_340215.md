    ```javascript
    test('TC-003 roundMoney rounds typical decimal value up', () => {
      expect(target.roundMoney(12.3456)).toBe(12.35);
    });
    test('TC-004 roundMoney rounds half‑up at .005', () => {
      expect(target.roundMoney(7.125)).toBe(7.13);
    });
    ```
